viewof transformation = Inputs.select(["Raw Proportion", , "Freeman-Tukey double arcsine transformed", "Log transformed", "Logit transformed",
"Arcsine square root transformed" ], {label: "Select Transformation"});
// Create a reactive dataset based on the selected transformation
transformedData = df.map(row => {
  if (transformation === "Log transformed") {
    return { ...row, effect_size: row.PLN };
    
  } else if (transformation === "Freeman-Tukey double arcsine transformed") {
    return { ...row, effect_size: row.PFT };
    
  } else if (transformation === "Logit transformed") {
    return { ...row, effect_size: row.PLO }; 
    
  } else if (transformation === "Arcsine square root transformed") {
    return { ...row, effect_size: row.PAS }; 
    
  } else {
    return { ...row, effect_size: row.OA_reported }; // Original data
  }
});
// Plot based on the selected transformation
Plot.plot({
  round: true,
  color: {legend: true},
  marks: [
    Plot.rectY(transformedData, Plot.binX({y2: "count"}, {x: "effect_size"})),
    Plot.ruleY([0])
  ], 
  y: {
    domain: [0, 45]
    }, 
  x: {
  label: "Effect Size"
}
})
